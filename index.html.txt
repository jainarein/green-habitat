    <script>
        const searchBtn = document.getElementById('searchBtn');
        const locationInput = document.getElementById('locationInput');
        const loader = document.getElementById('loader');
        const errorDiv = document.getElementById('error');
        const resultsDiv = document.getElementById('results');

        // Color definitions matching CSS
        const colors = {
            platinum: 'var(--platinum)',
            green: 'var(--green)',
            moderate: 'var(--moderate)',
            stress: 'var(--stress)'
        };

        // Determine badge details based on score
        function getCertification(score) {
            if (score >= 85) return { name: 'Platinum Peace Zone', color: colors.platinum, bg: 'rgba(251, 191, 36, 0.15)' };
            if (score >= 70) return { name: 'Green Certified', color: colors.green, bg: 'rgba(74, 222, 128, 0.15)' };
            if (score >= 55) return { name: 'Moderate Living', color: colors.moderate, bg: 'rgba(251, 146, 60, 0.15)' };
            return { name: 'Urban Stress Area', color: colors.stress, bg: 'rgba(239, 68, 68, 0.15)' };
        }

        // Animate Circular Progress Bar
        function animateProgress(targetScore, color) {
            const circularProgress = document.getElementById('circularProgress');
            const progressValue = document.getElementById('progressValue');
            
            circularProgress.style.setProperty('--score-color', color);
            
            let currentScore = 0;
            const speed = 20; // ms per frame
            
            progressValue.style.color = color;

            const interval = setInterval(() => {
                currentScore++;
                circularProgress.style.setProperty('--score', currentScore);
                progressValue.textContent = currentScore;

                if (currentScore >= targetScore) {
                    clearInterval(interval);
                }
            }, speed);
        }

        // Update individual parameter cards
        function updateParameter(id, data, themeColor) {
            const bar = document.getElementById(`bar-${id}`);
            const scoreText = document.getElementById(`score-${id}`);
            const indicator = document.getElementById(`ind-${id}`);

            // Handle data robustly (whether API sends object with 'score' or just a primitive number)
            const scoreValue = typeof data === 'object' ? (data.score || 0) : (data || 0);
            const isLive = typeof data === 'object' && data.is_live !== undefined ? data.is_live : true;

            // Animate bar width
            setTimeout(() => {
                bar.style.width = `${scoreValue}%`;
                bar.style.background = `linear-gradient(90deg, ${themeColor}88, ${themeColor})`;
            }, 300);

            scoreText.textContent = `${scoreValue}/100`;

            // Live/Mock Indicator
            if (isLive) {
                indicator.textContent = "Live";
                indicator.className = "data-indicator live";
            } else {
                indicator.textContent = "Mock";
                indicator.className = "data-indicator";
            }
        }

        // Main Fetch Logic
        searchBtn.addEventListener('click', async () => {
            const locName = locationInput.value.trim();
            if (!locName) return;

            // Reset UI
            resultsDiv.style.display = 'none';
            errorDiv.style.display = 'none';
            loader.style.display = 'block';
            
            // Reset Progress bars
            document.querySelectorAll('.bar-fill').forEach(bar => bar.style.width = '0%');
            document.getElementById('circularProgress').style.setProperty('--score', '0');
            document.getElementById('progressValue').textContent = '0';

            try {
                // Exact API call requested
                const response = await fetch(`http://127.0.0.1:8000/rate-area?location=${encodeURIComponent(locName)}`);
                
                if (!response.ok) {
                    throw new Error(`Location not found or server error (${response.status})`);
                }

                const data = await response.json();
                
                // --- Data Normalization ---
                // Adjusting to fallback gracefully depending on exact JSON keys your Python backend returns
                const score = data.peace_score || data.score || data.total_score || 0;
                const coords = data.coordinates || data.coords || "Coordinates unavailable";
                const locationLabel = data.location || data.name || locName;
                const params = data.parameters || data || {}; 

                const cert = getCertification(score);

                // Update Header
                document.getElementById('locName').textContent = locationLabel;
                document.getElementById('locCoords').textContent = coords;
                
                const badgeElement = document.getElementById('certBadge');
                badgeElement.textContent = cert.name;
                badgeElement.style.color = cert.color;
                badgeElement.style.background = cert.bg;
                badgeElement.style.border = `1px solid ${cert.color}`;

                // Update UI Layout
                loader.style.display = 'none';
                resultsDiv.style.display = 'block';

                // Trigger Animations
                animateProgress(score, cert.color);
                updateParameter('greenery', params.greenery || 0, cert.color);
                updateParameter('aqi', params.aqi || 0, cert.color);
                updateParameter('traffic', params.traffic || 0, cert.color);
                updateParameter('crowd', params.crowd || params.crowd_density || 0, cert.color);
                updateParameter('noise', params.noise || 0, cert.color);

            } catch (err) {
                loader.style.display = 'none';
                errorDiv.style.display = 'block';
                document.getElementById('errorMsg').textContent = err.message || "Unable to connect to the local API. Ensure it is running on port 8000.";
            }
        });

        // Trigger search on Enter key
        locationInput.addEventListener('keypress', (e) => {
            if (e.key === 'Enter') searchBtn.click();
        });
    </script>
